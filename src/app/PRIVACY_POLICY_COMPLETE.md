```
✅ REQUIRED - Parent phone

❌ Regular responses

❌ Academic preferences

```

```
✅ No public profiles
```

## 🔧 Technical Implementation

### Get Data Collection Config:

```typescript
import { getDataCollectionConfig } from './types/privacy-policy-configs';

const config = getDataCollectionConfig(AgeCategory.EARLY_LEARNER);

// Check what data is collected
console.log(config.personalInfo.fullName);        // false
console.log(config.personalInfo.nickname);        // true
console.log(config.parentInfo.required);          // true
console.log(config.assessmentData.simplifiedResponses); // true
```

---

### Get Data Sharing Config:

```typescript
import { getDataSharingConfig } from './types/privacy-policy-configs';

const sharing = getDataSharingConfig(AgeCategory.JUNIOR_LEARNER);

// Check sharing permissions
console.log(sharing.withParents.allowed);         // true
console.log(sharing.withParents.requirement);     // "recommended"
console.log(sharing.withSchools.requiresAgreement); // true
console.log(sharing.prohibited);                  // ["Advertisers", "Social platforms", ...]
```

---

### Get Data Retention Config:

```typescript
import { getDataRetentionConfig } from './types/privacy-policy-configs';

const retention = getDataRetentionConfig(AgeCategory.SENIOR_ADOLESCENT);

// Check retention policies
console.log(retention.inactivityPeriod);          // "24-36 months"
console.log(retention.deletionOnRequest);         // true
console.log(retention.deletionRequestedBy);       // ["Student", "School"]
console.log(retention.whatIsDeleted);             // ["All assessments", "All insights", ...]
```
